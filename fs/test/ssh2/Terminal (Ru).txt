###########################################################################################################################################################
########################################################## QNX Terminal Небольшой мануал ##################################################################
###########################################################################################################################################################

uname -a    Получение имени операционной системы.

############################################################################################################################################################

hwid        Информация о ревизии устройства и объемах памяти. 

############################################################################################################################################################

df -h       Вывод информации о свободном дисковом пространстве.

############################################################################################################################################################

showmem -S  Отображение информации о памяти RAM

############################################################################################################################################################

top -d      Просмотр информации об использовании системы (процент загрузки CPU и использовании RAM)

Для завершения работы утилиты  нажмите на клавиатуре клавиши Ctrl + С 

############################################################################################################################################################

dvfs_client
cat /dev/dvfs_client_1    Просмотр информации о температуре, частоте и загруке ядер CPU

############################################################################################################################################################

usb-dcd -v  Информация о подключенных usb устройствах.

############################################################################################################################################################

ping google.com   Утилита поможет проверить скорость и качество подключения к Интернету (в качестве примера выбрал google)

Для завершения работы утилиты  нажмите на клавиатуре клавиши Ctrl + С 

############################################################################################################################################################

screenshot -filename=/fs/usb0/scr.png   Скриншот текущего экрана, путь сохранения корневой каталог USB-флешки.

############################################################################################################################################################

ls -l /fs/images        Вывод списка содержимого каталога (в качестве примера содержимое раздела images)

############################################################################################################################################################

cat /fs/mp/Version.inf           Показать содержимое файла в терминале, в данном примере Version.inf

############################################################################################################################################################

dd if=/dev/hd0t182 of=/tmp/hd.img      Пример работы утилиты dd, копирования раздела диска с серийным номером в папку tmp на устройстве.    

############################################################################################################################################################

slay APP_SUM
slay fordhmi
slay HMI_AL         Убить интерфейс Sync, все 3 команды (будет черный экран)

mmrplay -a volume=15 -v screen: /fs/usb0/Parker.mp4        Использование встроенного видеоплеера mp4 синка и воспроизведение с usb клипа "Parker.mp4"

Для завершения работы утилиты  нажмите на клавиатуре клавиши Ctrl + С 

shutdown      перезагрузка apim (после перезагрузки будет работать в штатном режиме).

############################################################################################################################################################
#################################################################### Работа с ftp ##########################################################################
############################################################################################################################################################
Для полноценной работы (запись, удаление и изменение файлов) с разделом mp, нужно его смонтировать командой:

mount -uw /fs/mp
sync

После завершения манипуляций с файлами, вернуть в исходное состояние:

mount -ur /fs/mp
sync
sync
shutdown      перезагрузка apim, после перезагрузки изменения вступят в силу (если удалили или изменили что то важное, будет кирпич, аккуратнее!!!)
